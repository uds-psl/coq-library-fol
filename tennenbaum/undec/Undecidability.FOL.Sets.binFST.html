Definition is_eset (t : term') :=
  ∀ ¬ ($0 ∈' t`[↑]).

Definition is_adj (x y t : term') :=
  ∀ $0 ∈' t`[↑] ↔ $0 ∈' x`[↑] ∨ $0 ≡' y`[↑].

Definition sub' (x y : term') :=
  ∀ $0 ∈' x`[↑] → $0 ∈' y`[↑].

Definition ax_ext' :=
  ∀ ∀ sub' $1 $0 → sub' $0 $1 → $1 ≡' $0.

Definition ax_eset' :=
  ∃ is_eset $0.

Definition ax_adj' :=
  ∀ ∀ ∃ is_adj $2 $1 $0.

Definition binFST :=
  ax_ext' :: ax_eq_elem' :: ax_eset' :: ax_adj' :: nil.
