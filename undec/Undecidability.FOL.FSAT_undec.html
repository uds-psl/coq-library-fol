  Theorem FSAT_undec :
    undecidable FSAT.
  Proof.
    apply (undecidability_from_reducibility dPCPb_undec).
    apply (reduces_transitive dPCPb_to_BPCP).
    eapply reduces_transitive; [| apply reduction].
    apply (@FULL_TRAKHTENBROT_non_informative (Σrel 2)). left. exists tt. auto.
  Qed.

  Qed.
