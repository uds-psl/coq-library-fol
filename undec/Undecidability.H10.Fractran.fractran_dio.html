Section exp_diophantine.

  Notation power := (mscal mult 1).

  Fact power_expo x y : power x y = y^x.
  Proof.
    induction x as [ | x IHx ]; simpl.
    + rewrite power_0; auto.
    + rewrite power_S; f_equal; auto.
  Qed.


  Let exp_dio n i j : ùîªF (fun v => exp i (fun2vec j n v)).
  Proof.
    revert j i; induction n as [ | n IHn ]; intros j i.
    + simpl; dio auto.
    + by dio equiv (fun v => power (v j) (qs i) * exp (S i) (fun2vec (S j) n v)).
      abstract (intros v; simpl fun2vec; rewrite exp_cons, power_expo; auto).
  Defined.

  Let exp_dio_lift n i j : ùîªF (fun v => exp i (fun2vec j n v‚≠≥)).
  Proof.
    revert j i; induction n as [ | n IHn ]; intros j i.
    + simpl; dio auto.
    + by dio equiv (fun v => power (v (S j)) (qs i) * exp (S i) (fun2vec (S j) n v‚≠≥)).
      abstract (intros v; simpl fun2vec; rewrite exp_cons, power_expo; auto).
  Qed.


  Fact fractran_exp_diophantine n : ùîªF (fun ŒΩ => ps 1 * exp 1 (fun2vec 0 n ŒΩ)).
  Proof. dio auto. Show Proof. Defined.

  Fact fractran_exp_diophantine' n : ùîªF (fun ŒΩ => ps 1 * exp 2 (fun2vec 1 n ŒΩ‚≠≥)).
  Proof. eapply dio_fun_mult. dio auto. eapply exp_dio_lift. Qed.

  Fact fractran_exp_diophantine'' : ùîªF (fun ŒΩ => ŒΩ 1 * qs 1 ^ ŒΩ 2).
  Proof.
    by dio equiv (fun ŒΩ => ŒΩ 1 * power (ŒΩ 2) (qs 1)).

  Qed.
